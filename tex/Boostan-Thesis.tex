\fi

\@endpart}

    \ifnum \c@secnumdepth >\m@ne

        \vskip 20\p@
      \fi
    \fi
    \interlinepenalty\@M
    \Huge \bfseries \raggedleft{ #1}\par\nobreak
    \vskip 50\p@
  }}
